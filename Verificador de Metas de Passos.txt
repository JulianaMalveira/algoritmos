/* Decomposição:
Total de dias de caminhada: 03
Entrada: Quantidade de passos dia 01
Entrada: Quantidade de passos dia 02
Entrada: Quantidade de passos dia 03

Processamento: Total acumulado de passos dos 03 dias: soma

Regra (Condição): 

- Se a soma total de passos for igual ou superior a 10.000, o programa deve dar os parabéns pela meta atingida. 

- Caso contrário, deve incentivar a pessoa a caminhar mais.

Saída: 

Mensagens: 
- Parabéns pela meta atingida!
- Você precisa caminhar mais.

*/


programa {
  funcao inicio() {
    inteiro quantidadePassosDiarios[3][2] = {
      {1,0},
      {2,0},
      {3,0}

    }

     para (inteiro contador = 0; contador < 3; contador++) {
      escreva("Quantos passos você deu no dia " + quantidadePassosDiarios [contador][0])
      escreva("\n")
      
      leia(quantidadePassosDiarios[contador][1])
    }
escreva (totalAcumuladoDePassos(quantidadePassosDiarios)) 

  }

  funcao cadeia totalAcumuladoDePassos (inteiro quantidadePassosDiarios[][]) {
     inteiro somaDePassos = 0

     para (inteiro contador = 0; contador < 3; contador++) {
      somaDePassos = somaDePassos + quantidadePassosDiarios [contador][1]
         
  }  

   se (somaDePassos >= 10000) {
         retorne ("Parabéns pela meta atingida! Sua soma de passos foi: " + somaDePassos)

     }
      retorne ("Você precisa caminhar mais. Sua soma de passos foi: " + somaDePassos)
}
}
